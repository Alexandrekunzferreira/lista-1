{
    class Program
    {
        static void Main(string[] args)
        {
            //Ex.4  Apartir dos valores da base e altura de um triângulo, calcular e exibir sua área.
            
            //Declarando as Variáveis.
            double comprimento, altura, area;
            Console.WriteLine(" Digite o comprimento do triângulo");
            comprimento = Convert.ToDouble(Console.ReadLine());
            Console.WriteLine(" Digite o valor da altura do triângulo");
            altura = Convert.ToDouble(Console.ReadLine());
            // Obs: A área do triângulo é base (b) ou comprimento vezes a altura (h) dividido por 2.
            area = comprimento * altura / 2;
            Console.WriteLine(" A área do triângulo é: " + area );
            Console.ReadKey();
        }
    }

 Exercicio 7
{
    class Program
    {
        static void Main(string[] args)
        {
            //Ex.7 Calcular e exibir a tensão de um determinado circuito eletrônico apartir dos valores da resistência 
            //e corrente elétrica que serão digitados, utilize a lei de Ohm.                       
            Double corrente, resistencia, tensao;
            Console.WriteLine("Digite o valor da corrente");
            corrente = Convert.ToDouble(Console.ReadLine());
            Console.WriteLine(" Digite o valor da resistência");
            resistencia = Convert.ToDouble(Console.ReadLine());
            //Existe mais duas formula na lei de Ohm totalizando três fórmulas.
            //conforme a lei de Ohm a formula é V(tensão) é igual a I(corrente) vezes R(resistência).
            tensao = corrente * resistencia;
            Console.WriteLine("A tensão do equipamento é;" + tensao);
            Console.ReadKey();
        }
    }
}
 exercicio 8
{
    class Program
    {
        static void Main(string[] args)
        {
            //Ex.8 Entrar via teclado com o valor de uma temperatura em graus Celsus, calcular e exibir sua temperatura 
            //equivalente em graus Fahrenheit.
            //Declaração de variáveis.
            
            Double celsus, fahrenheit;
            Console.WriteLine("Digite o valor da temperatura;");
            celsus = Convert.ToDouble(Console.ReadLine());
            //Para se converter a temperatura em graus Celsus para Fahrenheit temos que utilizar uma fórmula: 
            //c(Celsus) dividido por 5 igual a F (Fahrenheit) menos 32 dividido por 9.
            fahrenheit = ((celsus) /(5) * (9) + (32)) ;
            Console.WriteLine(" A temperatura em Fahrenheit é; "  +  fahrenheit);
            Console.ReadKey();
